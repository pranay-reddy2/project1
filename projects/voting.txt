    <script>
        let counta=0;
        let countb=0;
        let e1=document.getElementById("BarA");
        let e2=document.getElementById("BarB");
        document.getElementById("A").addEventListener("click", function() {
            counta++;
            updateProgress();
        });

        document.getElementById("B").addEventListener("click", function() {
            countb++;
            updateProgress();
        });

        function updateProgress() {
            let totalVotes = counta + countb;
            if (totalVotes > 0) {
                let percentageA = (counta / totalVotes) * 100;
                let percentageB = (countb / totalVotes) * 100;

                e1.style.width = percentageA + "%";
                e2.style.width = percentageB + "%";

                e1.innerText = Math.round(percentageA) + "%";
                e2.innerText = Math.round(percentageB) + "%";
            }
        }

    </script>
